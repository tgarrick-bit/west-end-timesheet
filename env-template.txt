# Supabase Configuration
# Copy this content to a file named .env.local and fill in your actual values

# Your Supabase project URL
NEXT_PUBLIC_SUPABASE_URL=https://your-project-id.supabase.co

# Your Supabase anonymous/public key
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here

# Your Supabase service role key (for admin operations)
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# Note: The NEXT_PUBLIC_ prefix makes these variables available in the browser
# Only use the service role key on the server side for admin operations
